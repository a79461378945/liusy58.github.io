快速排序的描述
----
就我理解的快速排序，就是找哨兵，然后把小于它的值放到他的前面去，大于的值放到后面去，所以这个哨兵的位置就确定了从而以后就可以不用管他了。
```c++
int partion(vector<int>&a,int l,int r)
{
    int x=a[r];
    int i=l-1;
    int j=l;
    while(j<r)
    {
        if(a[j]<=x)
        {
            swap(a[++i],a[j]);
        }
        ++j;
    }
    swap(a[++i],a[r]);
    return i;
}
void quicksort(vector<int>&a,int l,int r)
{
    if(l>=r)
        return;
    int middle=partion(a, l, r);
    quicksort(a,l, middle-1);
    quicksort(a, middle+1, r);
    
}
```

如果基本有序的数组就用插入排序。

优化--针对有相同元素的快速排序

```c++
void partition(int A[], int p, int r, int &q, int &t)
{
	// A[q..t]中所有元素都等于主元素
	// A[p..q-1]中每个元素都小于主元素
	// A[t+1..r]中每个元素都大于主元素
	int x = A[r];
	q = p-1, t = r;
	for (int i = p; i < r; i++)
	{
		if (i >= t)
			break;
		while (A[i] > x && i < t)
			swap(A[--t], A[i]);
		if (A[i] < x)
			swap(A[++q], A[i]);
	}
	swap(A[t], A[r]);
	q++;
}
void quick_sort(int A[], int p, int r)
{
	if (p < r)
	{
		int q, t;
		partition(A, p, r, q, t);
		quick_sort(A, p, q - 1);
		quick_sort(A, t + 1, r);
	}
}

```
